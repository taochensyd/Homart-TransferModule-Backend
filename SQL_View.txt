CREATE VIEW [dbo].[Homart_JournalMemo_B1SLQuery] AS 
SELECT [JrnlMemo]
FROM OWTR
WHERE JrnlMemo LIKE 'WEB STOCK Transferno:%'
ORDER BY DocEntry DESC
OFFSET 0 ROWS
FETCH NEXT 1 ROWS ONLY;
